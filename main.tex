    \section{一个可能有用的HACK}
    
    \begin{frame}
        \frametitle{距离的人工调整}
        \vspace{-6cm} % 垂直距离的HACK
        \begin{itemize}
            \item 比如你觉得正文内容和标题离的太远
            \item 这样就可以减少了，vspace命令是垂直距离
            \item v 就是 vertical
            \item 参数值可以是正的(扩大)，也可以是负的(缩小)
            \item 水平距离，就是hspace，语法也一样
            \item h 就是 horizontal
        \end{itemize}
    \end{frame}
